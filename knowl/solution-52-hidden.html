Given: \(m= 5\,kg, \quad w= mg = 5\,kg\cdot9.8 \,m/s^{2} =49 \,N,\quad F=100 \,N.\)
  1. \begin{equation*} \sum F_{x}=F-N=0 \quad\Rightarrow\quad F=N=100\,N \end{equation*}
    \begin{equation*} \sum F_{y}=f-w=0\quad\Rightarrow\quad f=w=49\,N \end{equation*}
  2. \begin{equation*} f=\mu_{s}N \qquad\Rightarrow\quad N=\frac{f}{\mu_{s}}=\frac{49N}{0.4}=122\,N \end{equation*}

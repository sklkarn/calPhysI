\begin{equation*} N\cos\theta =mg \end{equation*}
\begin{equation*} N\sin\theta =\frac{mv^{2}}{r} =\frac{mr^{2}\omega^{2}}{r} = mr\omega^{2} = mR\sin\theta\omega^{2} \end{equation*}
\begin{equation*} \therefore N = mR\omega^{2} \end{equation*}
\begin{equation*} \text{or,}\quad mR\omega^{2} \cos\theta = mg \end{equation*}
\begin{equation*} \therefore \cos\theta = \frac{g}{R\omega^{2}} \end{equation*}
\(\because \cos\theta\leq 1, \)
\begin{equation*} \frac{g}{R\omega^{2}}\leq 1. \end{equation*}
\begin{equation*} \therefore \omega \leq \sqrt{\frac{g}{R}} \end{equation*}

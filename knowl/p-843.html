\begin{equation*} \therefore \quad \frac{m_{b}}{m_{d}} = \frac{\left[g\sin\theta +\mu_{s}g\cos\theta +a\right]}{\left[g\sin\theta -\mu_{k}g\cos\theta -a\right] } \end{equation*}

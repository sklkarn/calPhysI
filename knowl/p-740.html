\begin{equation*} F=ma=-\frac{GMm}{r^{2}} \end{equation*}
\begin{equation} \text{or,}\quad \ddot{r}-r\dot{\theta}^{2}=-\frac{GM}{r^{2}} \tag{6.2.3} \end{equation}
\begin{equation*} x=r\cos\theta, \qquad y=r\sin\theta \end{equation*}
\begin{equation*} \dot{x}=\dot{r}\cos\theta -r\sin\theta\dot{\theta}, \end{equation*}
\begin{equation*} \dot{y}=\dot{r}\sin\theta +r\cos\theta\dot{\theta} \end{equation*}
\begin{equation*} v^{2}=\dot{x}^{2}+\dot{y}^{2} = \dot{r}^{2}+r^{2}\dot{\theta}^{2} \end{equation*}
\begin{equation*} \ddot{x}=\ddot{r}\cos\theta -\dot{r}\dot{\theta}\sin\theta -\dot{r}\dot{\theta}\sin\theta -r\dot{\theta}^{2}\cos\theta-r\ddot{\theta}\sin\theta, \end{equation*}
\begin{equation*} \ddot{y}=\ddot{r}\sin\theta +\dot{r}\dot{\theta}\cos\theta +\dot{r}\dot{\theta}\cos\theta -r\dot{\theta}^{2}\sin\theta+r\ddot{\theta}\cos\theta \end{equation*}
Now,
\begin{align*} a\amp =\ddot{x}\cos\theta+\ddot{y}\sin\theta \\ \amp =(\ddot{r}\cos^{2}\theta -2\dot{r}\dot{\theta}\sin\theta\cos\theta -r\dot{\theta}^{2}\cos^{2}\theta-r\ddot{\theta}\sin\theta\cos\theta) \\ \amp +(\ddot{r}\sin^{2}\theta +2\dot{r}\dot{\theta}\sin\theta\cos\theta -r\dot{\theta}^{2}\sin^{2}\theta+r\ddot{\theta}\sin\theta\cos\theta) \\ \amp =\ddot{r}-r\dot{\theta}^{2} \end{align*}

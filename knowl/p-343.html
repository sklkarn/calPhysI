Given: \(m= 5kg, \quad w= mg = 5kg\cdot9.8 m/s^{2} =49N,\quad F=100 N.\)
  1. \begin{equation*} \sum F_{x}=F-N=0 \quad\Rightarrow\quad F=N=100N \end{equation*}
    \begin{equation*} \sum F_{y}=f-w=0\quad\Rightarrow\quad f=w=49N \end{equation*}
  2. \begin{equation*} f=\mu_{s}N \qquad\Rightarrow\quad N=\frac{f}{\mu_{s}}=\frac{49N}{0.4}=122N \end{equation*}
